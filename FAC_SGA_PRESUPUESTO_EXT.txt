SELECT 
  DC.COD_COMPANIA,
CASE 
   WHEN DM.NOM_MATERIAL='HIBRIDO' AND DC.COD_COMPANIA='00105'
   THEN '00090182_0'
   WHEN DM.NOM_MATERIAL='GUINEENSIS' AND DC.COD_COMPANIA='00105'
   THEN '00090189_0'
   WHEN DM.NOM_MATERIAL='HIBRIDO' AND DC.COD_COMPANIA='00106'
   THEN '00091110_0'
   WHEN DM.NOM_MATERIAL='GUINEENSIS' AND DC.COD_COMPANIA='00106'
   THEN '00091117_0'
   WHEN DM.NOM_MATERIAL='HIBRIDO' AND DC.COD_COMPANIA='00107'
   THEN '00094284_0'
   WHEN DM.NOM_MATERIAL='GUINEENSIS' AND DC.COD_COMPANIA='00107'
   THEN '00094234_0'
   WHEN DM.NOM_MATERIAL='HIBRIDO' AND DC.COD_COMPANIA='00111'
   THEN '00091110_0'
   WHEN DM.NOM_MATERIAL='GUINEENSIS' AND DC.COD_COMPANIA='00111'
   THEN '00091117_0'
   ELSE 'REVISAR'
END AS ID_SUBLIBRO_CAT_ERI,
  PPTO.ID_PLANTACION, 
  PPTO.FECHA_INICIO, 
  PPTO.CANTIDAD, 
  PPTO.CANTIDAD_PRESUPUESTO_INICIAL, 
  PPTO.GLCLASS, 
  --PPTO.FECHA_CREACION,
  --PPTO.FECHA_ACTUALIZACION, 
  DO.NOM_ORIGEN,
  DM.NOM_MATERIAL,
  DTD.NOM_TIPO_DOCUMENTO,
  DE.NOM_ESTADO,
  CASE WHEN DTC.NOM_TIPO_CLASIFICACION IS NULL THEN "FRUTA"
       ELSE  DTC.NOM_TIPO_CLASIFICACION
  END AS NOM_TIPO_CLASIFICACION,
  PPTO.ETL_TSTAMP
FROM FAC_SGA_PRESUPUESTO_EXT PPTO
LEFT JOIN
    DIM_COMPANIA DC ON PPTO.ID_COMPANIA = DC.ID_COMPANIA 
LEFT JOIN 
    DIM_PLANTACION DP ON PPTO.ID_PLANTACION = DP.ID_PLANTACION
LEFT JOIN
    DIM_SGA_ORIGEN DO ON PPTO.ID_ORIGEN = DO.ID_ORIGEN
LEFT JOIN
    DIM_SGA_MATERIAL DM ON PPTO.ID_MATERIAL = DM.ID_MATERIAL
LEFT JOIN
    DIM_SGA_TIPO_DOCUMENTO DTD ON PPTO.ID_TIPO_DOCUMENTO =DTD.ID_TIPO_DOCUMENTO
LEFT JOIN
    DIM_ESTADO DE ON PPTO.ID_ESTADO_CULTIVO = DE.ID_ESTADO
LEFT JOIN
    DIM_SGA_TIPO_CLASIFICACION DTC ON PPTO.ID_TIPO_CLASIFICACION = DTC.ID_TIPO_CLASIFICACION
