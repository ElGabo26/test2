-- 1) Fila de TRASPASO (saldo inicial) en enero
SELECT
    EF.CCIA,
    EF.ANIO_FISCAL,
    EF.UNDNEGOCIO,
    CONCAT(EF.ID_CUENTA,'_',EF.SUBLIBRO) AS ID_SUBLIBRO,
    CAST(CONCAT('20', EF.ANIO_FISCAL, '-01-01') AS DATE) AS FECHA,       -- Traspaso de saldos (enero)
    EF.TRASPASO_SALDO AS TRASPASO_SALDO,
    0 AS Valor,
    0 AS PPTO_Valor
FROM DDM_ERP.FAC_ESTRESULTADOS_ACT_ALL EF
WHERE
    EF.ANIO_FISCAL >= '20'
    AND EF.ANIO_FISCAL <= SUBSTR(YEAR(CURRENT_TIMESTAMP), 3, 4)

UNION ALL -- Para el Real
SELECT
    EF.CCIA,
    EF.ANIO_FISCAL,
    EF.UNDNEGOCIO,
    CONCAT(EF.ID_CUENTA,'_',EF.SUBLIBRO) AS ID_SUBLIBRO,
    CAST(CONCAT('20', EF.ANIO_FISCAL, '-', M.MES, '-01') AS DATE) AS FECHA,
    0 AS TRASPASO_SALDO,
    CASE M.MES
        WHEN '01' THEN EF.ENE_ACT
        WHEN '02' THEN EF.FEB_ACT
        WHEN '03' THEN EF.MAR_ACT
        WHEN '04' THEN EF.ABR_ACT
        WHEN '05' THEN EF.MAY_ACT
        WHEN '06' THEN EF.JUN_ACT
        WHEN '07' THEN EF.JUL_ACT
        WHEN '08' THEN EF.AGO_ACT
        WHEN '09' THEN EF.SEP_ACT
        WHEN '10' THEN EF.OCT_ACT
        WHEN '11' THEN EF.NOV_ACT
        WHEN '12' THEN EF.DIC_ACT
    END AS Valor,
    0 AS PPTO_Valor
FROM DDM_ERP.FAC_ESTRESULTADOS_ACT_ALL EF
-- tabla de meses
JOIN (
    SELECT '01' AS MES UNION ALL SELECT '02' UNION ALL SELECT '03' UNION ALL
    SELECT '04' UNION ALL SELECT '05' UNION ALL SELECT '06' UNION ALL
    SELECT '07' UNION ALL SELECT '08' UNION ALL SELECT '09' UNION ALL
    SELECT '10' UNION ALL SELECT '11' UNION ALL SELECT '12'
) AS M ON 1=1
WHERE
    EF.ANIO_FISCAL >= '20'
    AND EF.ANIO_FISCAL <= SUBSTR(YEAR(CURRENT_TIMESTAMP), 3, 4)

UNION ALL -- Para el PPTO
SELECT
    EF.CCIA,
    EF.ANIO_FISCAL,
    EF.UNDNEGOCIO,
    CONCAT(EF.ID_CUENTA,'_',EF.SUBLIBRO) AS ID_SUBLIBRO,
    CAST(CONCAT('20', EF.ANIO_FISCAL, '-', M.MES, '-01') AS DATE) AS FECHA,
    0 AS TRASPASO_SALDO,
    0 AS Valor,
    CASE M.MES
        WHEN '01' THEN EF.ENE_ACT
        WHEN '02' THEN EF.FEB_ACT
        WHEN '03' THEN EF.MAR_ACT
        WHEN '04' THEN EF.ABR_ACT
        WHEN '05' THEN EF.MAY_ACT
        WHEN '06' THEN EF.JUN_ACT
        WHEN '07' THEN EF.JUL_ACT
        WHEN '08' THEN EF.AGO_ACT
        WHEN '09' THEN EF.SEP_ACT
        WHEN '10' THEN EF.OCT_ACT
        WHEN '11' THEN EF.NOV_ACT
        WHEN '12' THEN EF.DIC_ACT
    END AS PPTO_Valor
FROM FAC_ESTRESULTADOS_PRESUP_ALL EF
-- tabla de meses
JOIN (
    SELECT '01' AS MES UNION ALL SELECT '02' UNION ALL SELECT '03' UNION ALL
    SELECT '04' UNION ALL SELECT '05' UNION ALL SELECT '06' UNION ALL
    SELECT '07' UNION ALL SELECT '08' UNION ALL SELECT '09' UNION ALL
    SELECT '10' UNION ALL SELECT '11' UNION ALL SELECT '12'
) AS M ON 1=1
WHERE
    EF.ANIO_FISCAL >= '20'
    AND EF.ANIO_FISCAL <= SUBSTR(YEAR(CURRENT_TIMESTAMP), 3, 4)
