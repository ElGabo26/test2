--DANEC
SELECT 
    '00101' AS COD_COMPANIA,
    DF.MES AS FECHA,
    DP.COD_CAT_CONTA,
    RTRIM(DP.NOM_CAT_CONTA) AS NOM_CAT_CONTA,
    CASE 
        WHEN DP.COD_CAT_CONTA = 'S295' -- FACTURACION REEMBOLSO SIN IVA
        THEN 'REEMBOLSO DE GASTOS' -- DISMINUIR EL GASTO

        WHEN DP.COD_CAT_CONTA IN ('SV25','S335','S337','SV48','S384','S385','SV18') -- ASISTENCIA TECNICA SERVICIOS ADM
        THEN 'OTROS INGRESOS' -- OTROS INGRESOS

        WHEN DP.COD_CAT_CONTA IN ('INSU','S291') -- OTRAS VENTAS
             OR DP.COD_TIPO_FAB IN ('011')  -- FILTO DE SERVICIOS
             OR DP.ID_PRODUCTO ='50214' -- GRASA #3 EN ESCAMAS D0090026032 IND
        THEN 'OTRAS VENTAS' -- OTRAS VENTAS

        WHEN DTD.cod_documento IN ('SE', 'S5', 'FC', 'CT', 'C5', 'VE', 'CS')
        THEN 'EXPORTACIONES'

        WHEN DP.COD_SCLASIF_MCIA IN ('AMM', 'EXP', 'IND', 'MAQ', 'NUA', 'OLN', 'LYA', 'PPR')
              AND DC.COD_CANAL='EXP' -- Solo frontera
        THEN 'FRONTERA' -- VENTA DE FRONTERA


        WHEN DP.COD_SCLASIF_MCIA ='LYA'
        THEN 'LIMPIEZA Y ASEO' -- LIMPIEZA Y ASEO

        WHEN (DP.COD_SCLASIF_MCIA IN ('AMM', 'EXP', 'IND', 'MAQ', 'NUA', 'OLN', 'PPR') 
              OR DP.COD_CAT_CONTA = 'PTMQ')  -- Incluir ALMENDRA DE PALMISTE H Y G ID 51554 73315
              AND DP.COD_FAMILIA='111'
        THEN 'MAQUILA' -- MAQUILA

        WHEN DP.COD_SCLASIF_MCIA IN ('AMM', 'EXP', 'IND', 'MAQ', 'NUA', 'OLN', 'PPR') 
        THEN 'NACIONAL' -- VENTA LOCAL

        ELSE 'OTRAS VENTAS' -- OTRAS VENTAS
    END AS CANAL_VENTA,
    SUM(
         COALESCE(CM.COSTOXKILO,0) * FVT.CANT_VENDIDA * COALESCE(NULLIF(FVT.FACTOR_PESO, 0), DP.FACTOR_PESO) + 
         COALESCE(CM.COSTOXKILO,0) * FVT.CANT_NOTA_CRED * COALESCE(NULLIF(FVT.FACTOR_PESO, 0), DP.FACTOR_PESO)+
         COALESCE(CM.COSTOXKILO,0) * FVT.CANT_NOTA_DEB * COALESCE(NULLIF(FVT.FACTOR_PESO, 0), DP.FACTOR_PESO)
        )
   AS COSTO,
   SUM(
      CASE 
         WHEN DP.COD_UNID_NEGOCIO IN ('LYA','B2C','B2B','OLN','EXP','MAQ','NUA','PPR')
         THEN(
            FVT.CANT_VENDIDA   * COALESCE(NULLIF(FVT.FACTOR_PESO,0), DP.FACTOR_PESO)
         +  FVT.CANT_NOTA_CRED* COALESCE(NULLIF(FVT.FACTOR_PESO,0), DP.FACTOR_PESO)
         +  FVT.CANT_NOTA_DEB * COALESCE(NULLIF(FVT.FACTOR_PESO,0), DP.FACTOR_PESO))
      ELSE 0
    END)/1000 AS CANTIDAD_MIXTA,
    SUM(FVT.VENTA_AUTOCONSUMO) AS VENTA
FROM 
    DDM_ERP.FAC_VENTA_TOTAL FVT
LEFT JOIN 
    DDM_ERP.DIM_PRODUCTO DP ON FVT.ID_PRODUCTO = DP.ID_PRODUCTO
LEFT JOIN 
    DDM_ERP.DIM_TIPO_DOCUMENTO DTD ON FVT.ID_DOCUMENTO = DTD.ID_DOCUMENTO
LEFT JOIN 
    DDM_ERP.DIM_CLIENTE DC ON FVT.ID_CLIENTE =  DC.ID_CLIENTE
LEFT JOIN 
    DDM_ERP.DIM_FECHA DF ON FVT.DATE_CONTA = DF.FECHA

LEFT JOIN
    (SELECT 
            CODITEM, 
            FECHA, 
            MAX(ABS(COSTOXKILO)) AS COSTOXKILO
        FROM 
            DDM_ERP.FAC_COSTO_MARGEN
        WHERE 
            COSTOXKILO <> 0
            AND DAY(FECHA)=1
        GROUP BY
            CODITEM, FECHA

        UNION ALL

        (
            WITH RankedData AS (
                SELECT 
                    CODITEM,
                    DATE_ADD((SELECT MAX(FECHA) FROM DDM_ERP.FAC_COSTO_MARGEN WHERE DAY(FECHA)=1), INTERVAL 1 MONTH) AS FECHA,
                    COSTOXKILO,
                    ROW_NUMBER() OVER (PARTITION BY CODITEM ORDER BY FECHA DESC) AS rn
                FROM 
                    DDM_ERP.FAC_COSTO_MARGEN
                WHERE 
                    COSTOXKILO <> 0
            )
            SELECT 
                CODITEM,
                FECHA,
                ABS(COSTOXKILO)
            FROM 
                RankedData
            WHERE 
                rn = 1
        )
     ) CM ON FVT.ID_PRODUCTO = CM.CODITEM
                  AND DF.MES = CM.FECHA

WHERE 
    FVT.ID_COMPANIA = '101'
    AND FVT.ID_CLIENTE <>'101'
    AND DC.DESC_TIPO_ID <>'NO APLICA' -- Cedula, Ruc y Pasaporte
    AND FVT.DATE_CONTA >= '2025-01-01'
GROUP BY
    DF.MES,
    DP.COD_CAT_CONTA,
    DP.NOM_CAT_CONTA,
    CANAL_VENTA


UNION ALL
-- AGRICOLAS
SELECT 
    CASE
        WHEN
             UN.COD_CIA = '00105' AND
             (SUBSTR(UN.COD_UNID_NEGOCIO,1,1) ='Q' OR
              UN.COD_UNID_NEGOCIO LIKE '%QDE%')
        THEN '001052'
        ELSE CONCAT('00',FVT.ID_COMPANIA)
    END AS COD_COMPANIA,
    DF.MES AS FECHA,
    DP.COD_CAT_CONTA,
    RTRIM(DP.NOM_CAT_CONTA) AS NOM_CAT_CONTA,
    CASE 
        WHEN 
             DP.COD_CAT_CONTA IN ('ARPG')
        THEN 'ACEITE DE PALMA GUINEENSIS'
        WHEN 
             DP.COD_CAT_CONTA IN ('ARPH')
        THEN 'ACEITE DE PALMA HIBRIDA'
        WHEN 
             DP.COD_CAT_CONTA IN ('ADPA')
        THEN 'ACEITE DE PALMISTE HIBRIDO'
        WHEN 
             DP.COD_CAT_CONTA IN ('ACPA')
        THEN 'ACEITE DE PALMISTE GUINEENSIS'
        WHEN 
             DP.COD_CAT_CONTA IN ('TDPA')
        THEN 'TORTA DE PALMISTE'
        WHEN 
             DP.COD_CAT_CONTA IN ('FPCS','FPSH','FPCH','FPCM')
        THEN 'FRUTO'
        WHEN 
             DP.COD_CAT_CONTA IN ('POLE')
        THEN 'POLEN'
        WHEN 
             DP.COD_CAT_CONTA IN ('PLV1','PLV3')
        THEN 'PLANTAS GUINEENSIS'
        WHEN 
             DP.COD_CAT_CONTA IN ('PLV2','PLV4')
        THEN 'PLANTAS HIBRIDAS'
        WHEN 
             DP.COD_CAT_CONTA IN ('SSGU','SGGU')
        THEN 'SEMILLAS GUINEENSIS'
        WHEN 
             DP.COD_CAT_CONTA IN ('SSHI','SGHI')
        THEN 'SEMILLAS HIBRIDAS'
        WHEN 
             DP.COD_CAT_CONTA IN ('S139')
        THEN 'MAQUILA FRUTO DE PALMA'
        WHEN 
             DP.COD_CAT_CONTA IN ('SV35','SV36')
        THEN 'SERVICIOS'
        WHEN 
             DP.COD_CAT_CONTA IN ('S295')
        THEN 'REEMBOLSO DE GASTOS'
        WHEN 
             DP.COD_CAT_CONTA IN ('SV25','S335','S337','SV48','SV18')
        THEN 'OTROS INGRESOS'
        WHEN 
             DP.COD_CAT_CONTA IN ('S142')
        THEN 'PREMIO CALIDAD DE ACEITE'
        ELSE 'OTRAS VENTAS' -- OTRAS VENTAS
    END AS CANAL_VENTA,
   0 AS COSTO,
    CASE 
        WHEN CANAL_VENTA IN ('POLEN','PLANTAS GUINEENSIS','TORTA DE PALMISTE','PLANTAS HIBRIDAS','SEMILLAS GUINEENSIS','SEMILLAS HIBRIDAS')
        THEN SUM(FVT.CANT_VENDIDA + FVT.CANT_NOTA_CRED + FVT.CANT_NOTA_DEB)
        WHEN CANAL_VENTA IN ('MAQUILA FRUTO DE PALMA')
        THEN SUM(FVT.VENTA_CAJ_REAL + FVT.CANT_NOTA_CRED + FVT.CANT_NOTA_DEB) -- Para corregir cantidad decimal
        WHEN CANAL_VENTA IN ('OTRAS VENTAS','SERVICIOS','OTROS INGRESOS','REEMBOLSO DE GASTOS','PREMIO CALIDAD DE ACEITE')
        THEN 0
        ELSE SUM(FVT.CANT_VENDIDA + FVT.CANT_NOTA_CRED + FVT.CANT_NOTA_DEB)/1000
    END AS CANTIDAD_MIXTA,
    SUM(FVT.VENTA_AUTOCONSUMO) AS VENTA
FROM DDM_ERP.FAC_VENTA_TOTAL FVT
LEFT JOIN DDM_ERP.DIM_PRODUCTO DP ON FVT.ID_PRODUCTO = DP.ID_PRODUCTO
LEFT JOIN DDM_ERP.DIM_CLIENTE DC ON FVT.ID_CLIENTE =  DC.ID_CLIENTE
LEFT JOIN DDM_ERP.DIM_FECHA DF ON FVT.DATE_CONTA = DF.FECHA
LEFT JOIN DDM_ERP.DIM_UNIDAD_NEGOCIO UN ON FVT.ID_UNID_NEGOCIO = UN.ID_UNID_NEGOCIO
WHERE 
    FVT.ID_COMPANIA IN ('105','106','107','111')
    AND FVT.ID_CLIENTE <> FVT.ID_COMPANIA
    AND FVT.VENTA_AUTOCONSUMO <> 0 -- Para no cargar valores en cero
    AND DC.DESC_TIPO_ID <>'NO APLICA' -- Quitar movimientos internos y mantener Cedula, Ruc y Pasaporte
    AND FVT.DATE_CONTA >= '2025-01-01'
GROUP BY
    COD_COMPANIA,
    DF.MES,
    DP.COD_CAT_CONTA,
    DP.NOM_CAT_CONTA,
    CANAL_VENTA
-- END AGRICOLAS


-- ESTADOS FINANCIEROS (CUENTA CONTABLE)
UNION ALL
SELECT
    EF.CCIA AS COD_COMPANIA,
    CAST(CONCAT('20', EF.ANIO_FISCAL, '-', M.MES, '-01') AS DATE) AS FECHA,
    'SV35' AS COD_CAT_CONTA,
    'FLETE EN EXPORTACION' AS NOM_CAT_CONTA,
    'OTRAS VENTAS' AS CANAL_VENTA,
    CASE M.MES
        WHEN '01' THEN EF.ENE_ACT
        WHEN '02' THEN EF.FEB_ACT
        WHEN '03' THEN EF.MAR_ACT
        WHEN '04' THEN EF.ABR_ACT
        WHEN '05' THEN EF.MAY_ACT
        WHEN '06' THEN EF.JUN_ACT
        WHEN '07' THEN EF.JUL_ACT
        WHEN '08' THEN EF.AGO_ACT
        WHEN '09' THEN EF.SEP_ACT
        WHEN '10' THEN EF.OCT_ACT
        WHEN '11' THEN EF.NOV_ACT
        WHEN '12' THEN EF.DIC_ACT
    END AS COSTO,
    0 AS CANTIDAD_MIXTA,
    0 AS VENTA
FROM DDM_ERP.FAC_ESTRESULTADOS_ACT_ALL EF
-- tabla de meses
JOIN (
    SELECT '01' AS MES UNION ALL SELECT '02' UNION ALL SELECT '03' UNION ALL
    SELECT '04' UNION ALL SELECT '05' UNION ALL SELECT '06' UNION ALL
    SELECT '07' UNION ALL SELECT '08' UNION ALL SELECT '09' UNION ALL
    SELECT '10' UNION ALL SELECT '11' UNION ALL SELECT '12'
) AS M ON 1=1
WHERE
    EF.ANIO_FISCAL >= '25'
	AND EF.ID_CUENTA = '00176320'
    AND EF.ANIO_FISCAL <= SUBSTR(YEAR(CURRENT_TIMESTAMP), 3, 4)
